## transition标签的用法
```html
<div id="app">
		<transition name="fade">	
			<div v-if="show">出现</div>
		</transition>
		<button @click="change">change</button>
	</div>
```
```css
.fade-leave-to,.fade-enter {
			opacity: 0;
		}
		.fade-leave-active,.fade-enter-active {
			transition: opacity 3s;
		}
```
```js
var vm = new Vue({
			el: "#app",
			data: {
				show: true
			},
			methods: {
				change: function() {
					this.show = !this.show;
				}
			}
		})
```
